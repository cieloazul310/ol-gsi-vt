import MdxMapLayout from '../../layout/mdxMapLayout';
import withoutAnno from '../../map/layers/without-anno';

export const meta = {
  title: '注記なし',
  layer: withoutAnno,
  layerId: 'without-anno',
};

**@cieloazul310/ol-gsi-vt**は、国土地理院が提供するベクトルタイルを OpenLayersで設定不要で手軽に表示するためのパッケージです。ベクトルタイル、最適化ベクトルタイルのそれぞれに通常、淡色の2種類ずつ、計4種類のプリセットレイヤを搭載しています。

## Example code

```ts
import { gsiOptVtLayer, gsiOptVtLayerExclude } from '@cieloazul310/ol-gsi-vt';

const withoutAnno = gsiOptVtLayer({
  layers: gsiOptVtLayerExclude(['Anno']),
});

export default withoutAnno;
```

export default ({ children }) => <MdxMapLayout meta={meta}>{children}</MdxMapLayout>;
